-- Создание базы данных
CREATE DATABASE IF NOT EXISTS internet_access_manager
CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE internet_access_manager;

-- Таблица пользователей
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL COMMENT 'Хранится в открытом виде (только для разработки!)',
    email VARCHAR(100) NOT NULL UNIQUE,
    role ENUM('Administrator', 'Technician', 'Teacher') NOT NULL DEFAULT 'Teacher',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    last_login_at DATETIME NULL,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    INDEX idx_role (role),
    INDEX idx_is_active (is_active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Таблица аудиторий
CREATE TABLE IF NOT EXISTS audiences (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    building VARCHAR(100) NULL,
    floor INT NULL,
    description TEXT NULL,
    current_status ENUM('FullAccess', 'NoAccess', 'SafeMode') NOT NULL DEFAULT 'FullAccess',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    INDEX idx_status (current_status),
    INDEX idx_building (building),
    INDEX idx_is_active (is_active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Таблица компьютеров
CREATE TABLE IF NOT EXISTS computers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    ip_address VARCHAR(45) NOT NULL,
    mac_address VARCHAR(17) NULL,
    audience_id INT NOT NULL,
    current_status ENUM('FullAccess', 'NoAccess', 'SafeMode') NOT NULL DEFAULT 'FullAccess',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    FOREIGN KEY (audience_id) REFERENCES audiences(id) ON DELETE CASCADE,
    INDEX idx_ip_address (ip_address),
    INDEX idx_audience_id (audience_id),
    INDEX idx_status (current_status),
    INDEX idx_is_active (is_active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Таблица назначенных аудиторий
CREATE TABLE IF NOT EXISTS assigned_audiences (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    audience_id INT NOT NULL,
    assigned_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    assigned_by_user_id INT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (audience_id) REFERENCES audiences(id) ON DELETE CASCADE,
    FOREIGN KEY (assigned_by_user_id) REFERENCES users(id),
    UNIQUE INDEX unique_user_audience (user_id, audience_id),
    INDEX idx_user_id (user_id),
    INDEX idx_audience_id (audience_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Таблица логов действий
CREATE TABLE IF NOT EXISTS action_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    action VARCHAR(50) NOT NULL,
    description TEXT NULL,
    audience_id INT NULL,
    computer_id INT NULL,
    old_status VARCHAR(20) NULL,
    new_status VARCHAR(20) NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    ip_address VARCHAR(45) NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (audience_id) REFERENCES audiences(id) ON DELETE SET NULL,
    FOREIGN KEY (computer_id) REFERENCES computers(id) ON DELETE SET NULL,
    INDEX idx_user_id (user_id),
    INDEX idx_audience_id (audience_id),
    INDEX idx_computer_id (computer_id),
    INDEX idx_created_at (created_at),
    INDEX idx_action (action)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Таблица блокировки системы
CREATE TABLE IF NOT EXISTS system_locks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    is_locked BOOLEAN NOT NULL DEFAULT FALSE,
    locked_by_user_id INT NULL,
    locked_at DATETIME NULL,
    reason TEXT NULL,
    FOREIGN KEY (locked_by_user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_is_locked (is_locked)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Представление для удобного просмотра аудиторий с количеством компьютеров
CREATE OR REPLACE VIEW view_audiences_with_stats AS
SELECT 
    a.*,
    COUNT(c.id) as computers_count,
    SUM(CASE WHEN c.current_status = 'FullAccess' THEN 1 ELSE 0 END) as computers_full_access,
    SUM(CASE WHEN c.current_status = 'NoAccess' THEN 1 ELSE 0 END) as computers_no_access,
    SUM(CASE WHEN c.current_status = 'SafeMode' THEN 1 ELSE 0 END) as computers_safe_mode
FROM audiences a
LEFT JOIN computers c ON c.audience_id = a.id AND c.is_active = TRUE
WHERE a.is_active = TRUE
GROUP BY a.id;

-- Представление для просмотра назначенных аудиторий с именами
CREATE OR REPLACE VIEW view_assigned_audiences AS
SELECT 
    aa.*,
    u.username as user_name,
    a.name as audience_name,
    a.building,
    a.floor,
    admin.username as assigned_by_user_name
FROM assigned_audiences aa
JOIN users u ON u.id = aa.user_id
JOIN audiences a ON a.id = aa.audience_id
JOIN users admin ON admin.id = aa.assigned_by_user_id;

-- Процедура для смены статуса аудитории (массовое обновление компьютеров)
DELIMITER $$
CREATE PROCEDURE sp_change_audience_status(
    IN p_audience_id INT,
    IN p_new_status ENUM('FullAccess', 'NoAccess', 'SafeMode'),
    IN p_user_id INT,
    IN p_ip_address VARCHAR(45)
)
BEGIN
    DECLARE v_old_status VARCHAR(20);
    DECLARE v_audience_name VARCHAR(100);
    
    -- Получаем старый статус и название аудитории
    SELECT current_status, name INTO v_old_status, v_audience_name
    FROM audiences WHERE id = p_audience_id;
    
    -- Обновляем статус аудитории
    UPDATE audiences 
    SET current_status = p_new_status 
    WHERE id = p_audience_id;
    
    -- Обновляем статус всех компьютеров в аудитории
    UPDATE computers 
    SET current_status = p_new_status 
    WHERE audience_id = p_audience_id AND is_active = TRUE;
    
    -- Логируем действие
    INSERT INTO action_logs (user_id, action, description, audience_id, old_status, new_status, ip_address)
    VALUES (p_user_id, 'ChangeAudienceStatus', 
            CONCAT('Changed audience "', v_audience_name, '" status from ', v_old_status, ' to ', p_new_status),
            p_audience_id, v_old_status, p_new_status, p_ip_address);
END$$
DELIMITER ;
